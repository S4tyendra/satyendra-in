import van from "vanjs-core"

const { div } = van.tags

/**
 * Lazy loading component wrapper using IntersectionObserver
 * @param {Function} importFn - Dynamic import function that returns a component module
 * @param {string} placeholderClass - CSS classes for the placeholder element
 * @returns {HTMLElement} Placeholder element that will be replaced with the actual component
 */
export const LazyComponent = (importFn, placeholderClass = "") => {
    const container = div({ class: placeholderClass })
    const loaded = van.state(false)

    const observer = new IntersectionObserver(
        async (entries) => {
            if (entries[0].isIntersecting && !loaded.val) {
                loaded.val = true
                observer.disconnect()

                try {
                    const module = await importFn()
                    const Component = module.default || module[Object.keys(module)[0]]

                    const component = Component()
                    container.replaceWith(component)
                } catch (error) {
                    console.error("Failed to load component:", error)
                }
            }
        },
        {
            rootMargin: "50px" 
        }
    )

    observer.observe(container)

    return container
}

export default LazyComponent

import van from "vanjs-core";
import { resolvePath } from "../paths.js";

const { div, span, input, kbd, ul, li, a } = van.tags;

const COMMANDS = [
    { name: "help", desc: "Show available commands" },
    { name: "ls", desc: "List directory contents" },
    { name: "cd", desc: "Change directory" },
    { name: "clear", desc: "Clear terminal output" },
];

const NAV_SHORTCUTS = [
    { name: "..", desc: "Go up one level" },
    { name: "~", desc: "Go to home" },
];

let outputCounter = 0;

export const TerminalBreadcrumb = () => {
    const currentPath = van.state(window.location.pathname);
    const inputValue = van.state("");
    const suggestions = van.state([]);
    const selectedIndex = van.state(0);
    const isExpanded = van.state(false);
    const terminalOutput = van.state([]);
    const isFocused = van.state(false);
    const hoveredDir = van.state(null);
    let hoverTimeout;

    // Handle URL changes
    const updatePath = () => {
        currentPath.val = window.location.pathname;
    };

    window.addEventListener("popstate", updatePath);

    const focusInput = () => document.getElementById("terminal-input")?.focus();
    const blurInput = () => document.getElementById("terminal-input")?.blur();

    // Global "/" key to focus terminal
    window.addEventListener("keydown", (e) => {
        const terminalInput = document.getElementById("terminal-input");
        if (e.key === "/" && document.activeElement !== terminalInput && !e.ctrlKey && !e.metaKey) {
            const tag = document.activeElement?.tagName?.toLowerCase();
            if (tag !== "input" && tag !== "textarea") {
                e.preventDefault();
                terminalInput?.focus();
            }
        }
    });

    const wrapHistory = (type) => {
        const original = history[type];
        return function (...args) {
            const result = original.apply(this, args);
            updatePath();
            return result;
        };
    };
    history.pushState = wrapHistory("pushState");
    history.replaceState = wrapHistory("replaceState");

    const navigate = (path) => {
        history.pushState({}, "", path);
        window.dispatchEvent(new PopStateEvent('popstate'));
        inputValue.val = "";
        refreshSuggestions();
        selectedIndex.val = 0;
    };

    const getChildren = () => {
        const resolved = resolvePath(currentPath.val);
        if (resolved && resolved.node && resolved.node.children) {
            return Object.entries(resolved.node.children).map(([name, node]) => ({
                name,
                type: node.type,
                path: node.path
            }));
        }
        return [];
    };

    const classifyInput = (value) => {
        const trimmed = value.replace(/^\s+/, "");
        if (!trimmed) return { mode: "default", query: "" };
        if (trimmed.startsWith("cd ")) {
            return { mode: "cd", query: trimmed.slice(3).trimStart() };
        }
        return { mode: "default", query: trimmed };
    };

    const filterMatches = (items, query, key = "name") => {
        if (!query) return items;
        const q = query.toLowerCase();
        return items
            .map((item) => {
                const target = (item[key] || "").toLowerCase();
                const idx = target.indexOf(q);
                if (idx === -1) return null;
                return { item, score: idx };
            })
            .filter(Boolean)
            .sort((a, b) => a.score - b.score || (a.item[key] || "").localeCompare(b.item[key] || ""))
            .map(({ item }) => item);
    };

    const buildSuggestions = (mode, query) => {
        const children = getChildren();
        const dirItems = children
            .filter((child) => child.type === "dir")
            .map((child) => ({ ...child, desc: child.desc || "directory" }));
        const fileItems = children
            .filter((child) => child.type === "file")
            .map((child) => ({ ...child, desc: child.desc || "file" }));

        const commandItems = COMMANDS.map((cmd) => ({ ...cmd, type: "cmd" }));
        const navItems = NAV_SHORTCUTS.map((shortcut) => ({ ...shortcut, type: "nav" }));

        if (mode === "cd") {
            return [
                ...filterMatches(dirItems, query),
                ...filterMatches(navItems, query),
            ];
        }

        const commandMatches = filterMatches(commandItems, query);
        const navMatches = filterMatches(navItems, query);
        const dirMatches = filterMatches(dirItems, query);
        const fileMatches = filterMatches(fileItems, query);

        return [
            ...commandMatches,
            ...navMatches,
            ...dirMatches,
            ...fileMatches,
        ];
    };

    const refreshSuggestions = () => {
        const { mode, query } = classifyInput(inputValue.val);
        suggestions.val = buildSuggestions(mode, query);
        selectedIndex.val = suggestions.val.length ? 0 : 0;
    };

    const clearOutput = () => {
        terminalOutput.val = [];
        if (!isFocused.val) {
            isExpanded.val = false;
        }
    };

    const shouldShowPanel = () => isFocused.val || inputValue.val.length > 0 || terminalOutput.val.length > 0 || isExpanded.val;

    const pushOutput = (entry) => {
        isExpanded.val = true;
        terminalOutput.val = [...terminalOutput.val, { id: ++outputCounter, ...entry }].slice(-6);
    };

    const executeCommand = (cmd) => {
        const trimmed = cmd.trim();

        if (!trimmed) return;

        // handle navigation shortcuts
        if (trimmed === ".." || trimmed === "cd ..") {
            const parts = currentPath.val.split("/").filter(Boolean);
            parts.pop();
            const newPath = "/" + parts.join("/");
            navigate(newPath || "/");
            return;
        }

        if (trimmed === "~" || trimmed === "cd ~") {
            navigate("/");
            return;
        }

        if (trimmed.startsWith("cd ")) {
            const target = trimmed.slice(3).trim();
            const child = getChildren().find(c => c.name === target && c.type === "dir");
            if (child) navigate(child.path);
            return;
        }

        if (trimmed === "ls") {
            const items = getChildren();
            if (items.length > 0) {
                pushOutput({ type: "ls", label: "ls", items });
            } else {
                pushOutput({ type: "text", label: "ls", text: "(empty directory)" });
            }
            inputValue.val = "";
            refreshSuggestions();
            return;
        }

        if (trimmed === "help") {
            pushOutput({
                type: "help",
                label: "help",
                items: [
                    { cmd: "help", desc: "Show commands" },
                    { cmd: "ls", desc: "List current directory" },
                    { cmd: "clear", desc: "Clear terminal output" },
                    { cmd: ".. / cd ..", desc: "Go up one level" },
                    { cmd: "~ / cd ~", desc: "Go home" },
                    { cmd: "<dir>", desc: "Jump directly" },
                    { cmd: "Tab", desc: "Pick suggestion" },
                    { cmd: "Shift+Tab", desc: "Previous suggestion" },
                    { cmd: "Esc", desc: "Close panel" },
                ]
            });
            inputValue.val = "";
            refreshSuggestions();
            return;
        }

        if (trimmed === "clear") {
            inputValue.val = "";
            clearOutput();
            refreshSuggestions();
            return;
        }

        // direct directory navigation
        const dir = getChildren().find(c => c.name === trimmed);
        if (dir && dir.path) navigate(dir.path);
    };

    const handleInput = (e) => {
        inputValue.val = e.target.value;
        if (inputValue.val.length && !isExpanded.val) {
            isExpanded.val = true;
        }
        refreshSuggestions();
    };

    const applySuggestion = (item, { fillOnly = false } = {}) => {
        if (!item) return;

        const context = classifyInput(inputValue.val).mode;

        if (item.type === "cmd") {
            inputValue.val = item.name + (item.name === "cd" ? " " : "");
            refreshSuggestions();
            if (!fillOnly) executeCommand(item.name);
            else focusInput();
            return;
        }

        if (item.type === "nav") {
            const value = context === "cd" ? `cd ${item.name}` : item.name;
            if (fillOnly) {
                inputValue.val = value;
                refreshSuggestions();
                focusInput();
            } else {
                executeCommand(value);
            }
            return;
        }

        if (item.type === "dir") {
            if (fillOnly) {
                inputValue.val = context === "cd" ? `cd ${item.name}` : item.name;
                refreshSuggestions();
                focusInput();
            } else if (item.path) {
                navigate(item.path);
            }
            return;
        }

        if (item.type === "file") {
            if (fillOnly) {
                inputValue.val = item.name;
                refreshSuggestions();
                focusInput();
            } else if (item.path) {
                navigate(item.path);
            }
        }
    };

    const handleKeyDown = (e) => {
        if (e.key === "Escape") {
            e.preventDefault();
            if (terminalOutput.val.length) {
                clearOutput();
            } else if (inputValue.val) {
                inputValue.val = "";
                refreshSuggestions();
                if (!isFocused.val) isExpanded.val = false;
            } else {
                isExpanded.val = false;
                blurInput();
            }
            return;
        }

        if (e.key === "Tab") {
            e.preventDefault();
            if (!suggestions.val.length) return;
            applySuggestion(suggestions.val[selectedIndex.val], { fillOnly: true });
            return;
        }

        if (e.key === "ArrowDown") {

            if (suggestions.val.length) {

            }
            return;
        }

        if (e.key === "ArrowUp") {
            e.preventDefault();
            if (suggestions.val.length) {
                selectedIndex.val = (selectedIndex.val - 1 + suggestions.val.length) % suggestions.val.length;
            }
            return;
        }

        if (e.key === "Enter") {
            e.preventDefault();
            const typed = inputValue.val.trim();

            if (!typed && selected) {
                applySuggestion(selected);
                return;
            }

            if (selected && typed && selected.name.toLowerCase() === typed.toLowerCase()) {
                executeCommand(selected.name);
                return;
            }

            executeCommand(typed);
            return;
        }
    };

        isFocused.val = true;
        if (!isExpanded.val) isExpanded.val = true;
        refreshSuggestions();
    };

                isExpanded.val = false;
            }
        }, 120);

                    return span({ class: "flex items-center" },

                        span({
                            class: `px-0.5 rounded cursor-pointer transition-colors text-[11px] ${isLast ? 'text-[#79c0ff] font-medium' : 'text-[#79c0ff]/70 hover:text-[#79c0ff] hover:bg-white/5'}`,
                            onmouseenter: (e) => {
                                clearTimeout(hoverTimeout);
                                if (crumb.node.children) {
                                    const rect = e.target.getBoundingClientRect();
                                    hoveredDir.val = { node: crumb.node, rect, id: crumb.path };
                                }
                            },
                            onmouseleave: () => {
                                hoverTimeout = setTimeout(() => hoveredDir.val = null, 200);
                            },
                            onclick: () => navigate(crumb.node.path)
                        }, shortName)
                    );
                })
            );
        };
    };

    const Popover = () => {
        return div({
            class: () => `fixed z-50 bg-[#0d1117]/90 backdrop-blur-md border border-[#30363d] rounded-md shadow-2xl p-2 min-w-[180px] transition-all duration-200 ease-out transform ${hoveredDir.val ? 'opacity-100 translate-y-0 visible' : 'opacity-0 translate-y-2 invisible'}`,
            style: () => {
                if (!hoveredDir.val) return "top: 0; left: 0;";
                const { rect } = hoveredDir.val;
                return `top: ${rect.bottom + 10}px; left: ${rect.left}px;`;
            },
            onmouseenter: () => clearTimeout(hoverTimeout),
            onmouseleave: () => hoveredDir.val = null,
        },
            () => {
                if (!hoveredDir.val) return [];
                const { node } = hoveredDir.val;
                if (!node.children) return span({ class: "text-zinc-500 text-xs px-2 italic" }, "No subdirectories");

                return ul({ class: "flex flex-col gap-1" },

                        li(
                            a({
                                class: "block px-3 py-1.5 text-sm text-zinc-300 hover:bg-[#1f6feb]/20 hover:text-[#58a6ff] rounded transition-colors cursor-pointer flex items-center gap-2",
                                onclick: (e) => {
                                    e.stopPropagation();
                                    navigate(childNode.path);
                                    hoveredDir.val = null;
                                }
                            },
                                span({ class: childNode.type === 'dir' ? "text-blue-400" : "text-zinc-400" }, childNode.type === 'dir' ? "üìÅ" : "üìÑ"),
                                name
                            )
                        )
                    )
                );
            }
        );
    };

    const renderOutputItem = (item) => {
        const header = div({ class: "px-3 py-1.5 text-[10px] uppercase tracking-widest text-zinc-500 border-b border-[#1f232e]" }, `$ ${item.label || ""}`);

        if (item.type === "ls") {

                        div({
                            class: `flex items-center gap-2 px-2 py-1.5 rounded cursor-pointer transition-colors text-xs ${entry.type === 'dir' ? 'text-blue-400 hover:bg-blue-500/10' : 'text-zinc-300 hover:bg-white/5'}`,
                            onclick: () => entry.path && navigate(entry.path)
                        },
                            span(entry.type === 'dir' ? 'üìÅ' : 'üìÑ'),

        }

export const resolvePath = (currentPath) => {
    // Simple helper to find the node in the fileSystem based on the URL path
    // This assumes paths match the structure in fileSystem
    if (currentPath === "/" || currentPath === "") return { node: fileSystem["~"], breadcrumbs: [{ name: "~", node: fileSystem["~"] }] };

    const parts = currentPath.split("/").filter(p => p);
    let currentNode = fileSystem["~"];
    const breadcrumbs = [{ name: "~", node: currentNode }];

    for (const part of parts) {
        if (currentNode.children && currentNode.children[part]) {
            currentNode = currentNode.children[part];
            breadcrumbs.push({ name: part, node: currentNode });
        } else {
            // Path not found in our defined structure, maybe return null or best effort
            return null;
        }
    }
    return { node: currentNode, breadcrumbs };
};

import van from "vanjs-core"

const getInitialTheme = () => {
  if (typeof localStorage !== 'undefined' && 'theme' in localStorage) {
    return localStorage.theme === 'dark'
  }
  if (typeof window !== 'undefined') {
    return window.matchMedia('(prefers-color-scheme: dark)').matches
  }
  return false
}

export const isDark = van.state(getInitialTheme())

if (isDark.val) {
  document.documentElement.classList.add('dark')
} else {
  document.documentElement.classList.remove('dark')
}

van.derive(() => {
  if (isDark.val) {
    document.documentElement.classList.add('dark')
    localStorage.theme = 'dark'
  } else {
    document.documentElement.classList.remove('dark')
    localStorage.theme = 'light'
  }
})
